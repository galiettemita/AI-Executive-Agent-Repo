1.3 Data Flow Graph — V10 Extended Pipeline
1.  INGRESS [WhatsApp/iMessage Webhook]
    Raw message + attachments received
 
2.  GATEWAY.normalize()
    Channel parsing, voice transcription, attachment upload to S3
 
3.  GATEWAY.multimodal.analyze() [NEW — Phase 1]
    Vision analysis, audio feature extraction, multimodal fusion
    Output: Enriched MessageEnvelope with attachments[], audio_features, multimodal_context
 
4.  BRAIN.fastpath.check() [NEW — Phase 4]
    IF complexity=SIMPLE AND confidence≥0.92 AND cached_answer valid:
      → SKIP steps 5-8, jump to step 9 with cached response (target <500ms)
    IF complexity=SIMPLE AND confidence≥0.92 AND no cache:
      → Execute single skill directly, skip reasoning loop (target <800ms)
 
5.  BRAIN.classify()
    Intent classification with EQ context, user profile from SOUL.md/USER.md
 
6.  BRAIN.decompose()
    Task graph generation with dependency DAG
 
7.  CONTROL.gate()
    ExecutionGate PDP: OPA policies, autonomy, budget, approval, wallet check
 
8.  EXECUTOR.execute() [per-skill, parallelized]
    Tool calls, browser automation, payments, federation negotiations
 
9.  BRAIN.aggregate()
    Result merging, EQ-calibrated response generation, tone adaptation
 
10. GATEWAY.format()
    Channel formatting (WhatsApp markdown, iMessage rich links), TTS if voice
 
11. EGRESS [WhatsApp/iMessage API]
    Delivery, 24-hour window tracking, template fallback if outside window

INSTANT:   0 LLM calls for reasoning. Pattern match → single skill → template. Target <500ms.
SHALLOW:   1 LLM call (PLANNER only). No CRITIC/REFLECTOR. Target <2s.
STANDARD:  Full V9 loop (PLANNER → EXECUTOR → CRITIC → REFLECTOR). Target <6s.
DEEP:      Full loop + multi-agent council/delegation. Target <20s.

2.1.1 New Endpoints
YAML
POST /v1/webhooks/whatsapp/media
  Description: Receives WhatsApp media webhook (images, audio, video, documents)
  Auth: Webhook signature verification (HMAC-SHA256)
  Request: WhatsApp media webhook payload
  Response: 202 Accepted
  Idempotency: channel_message_id unique constraint
  Processing:
    1. Download media from WhatsApp CDN (URL valid ~14 days)
    2. Upload to S3 with workspace_id prefix
    3. Create message_attachments record (status=pending)
    4. Enqueue attachment analysis job
    5. Attach analysis results to MessageEnvelope before Brain handoff
 
POST /v1/attachments/{attachment_id}/analyze
  Description: Trigger or retry analysis of a specific attachment
  Auth: UserJWT
  Request: { force_reanalysis: boolean }
  Response: { attachment_id, analysis_status, analysis_result }
  Error Codes:
    ATTACHMENT_NOT_FOUND (404)
    ANALYSIS_IN_PROGRESS (409)
    UNSUPPORTED_MEDIA_TYPE (415)
    ATTACHMENT_TOO_LARGE (413, max 20MB)
 
GET /v1/attachments/{attachment_id}
  Description: Retrieve attachment metadata and analysis result
  Auth: UserJWT
  Response: MessageAttachment object with analysis_result

-- UUIDv7 generation function (RFC 9562 compliant)
CREATE OR REPLACE FUNCTION uuidv7_generate() RETURNS UUID AS $$
DECLARE
  unix_ts_ms BIGINT;
  uuid_bytes BYTEA;
BEGIN
  unix_ts_ms := (EXTRACT(EPOCH FROM clock_timestamp()) * 1000)::BIGINT;
  uuid_bytes := decode(replace(gen_random_uuid()::TEXT, '-', ''), 'hex');
  uuid_bytes := set_byte(uuid_bytes, 0, (unix_ts_ms >> 40)::INT & 255);
  uuid_bytes := set_byte(uuid_bytes, 1, (unix_ts_ms >> 32)::INT & 255);
  uuid_bytes := set_byte(uuid_bytes, 2, (unix_ts_ms >> 24)::INT & 255);
  uuid_bytes := set_byte(uuid_bytes, 3, (unix_ts_ms >> 16)::INT & 255);
  uuid_bytes := set_byte(uuid_bytes, 4, (unix_ts_ms >> 8)::INT & 255);
  uuid_bytes := set_byte(uuid_bytes, 5, unix_ts_ms::INT & 255);
  uuid_bytes := set_byte(uuid_bytes, 6, (get_byte(uuid_bytes, 6) & 15) | 112); -- version 7
  uuid_bytes := set_byte(uuid_bytes, 8, (get_byte(uuid_bytes, 8) & 63) | 128); -- variant 10
  RETURN encode(uuid_bytes, 'hex')::UUID;
END;
$$ LANGUAGE plpgsql VOLATILE;

4.2 AttachmentAnalysisWorkflow (Phase 1)
YAML
ID: attachment-{attachment_id}
Task Queue: multimodal-analysis
Execution Timeout: 60s
Idempotency: workspace_id + ingress_turn_id + file_hash_sha256
 
Activities:
  DownloadMedia:
    Timeout: 15s
    Retry: 3 attempts, backoff 1s → 2s → 4s
    Input: { media_url, workspace_id }
    Output: { s3_url, file_size_bytes, file_hash }
    Compensation: Delete S3 object
 
  ClassifyAttachment:
    Timeout: 5s
    Input: { mime_type, file_size_bytes }
    Output: { attachment_type, analysis_strategy }
    Deterministic: Pure function, no side effects
 
  AnalyzeContent:
    Timeout: 30s
    Retry: 2 attempts, backoff 2s → 4s
    Input: { s3_url, attachment_type, analysis_strategy }
    Output: { analysis_result, embedding }
    Logic:
      IF image → call vision LLM (claude-sonnet-4-5 vision)
      IF audio → extract audio features + transcribe
      IF video → extract keyframes (1/5s, max 20) → batch vision analysis
      IF document → extract text, chunk, embed
    Guardrail: No facial recognition. Redact PII before storage.
 
  StoreResults:
    Timeout: 5s
    Input: { attachment_id, analysis_result, embedding }
    Output: { updated_attachment }
    Side effect: UPDATE message_attachments SET analysis_status='completed'
 
  FuseModalities:
    Timeout: 5s
    Input: { ingress_turn_id, all_attachment_analyses, audio_features, text_content }
    Output: { multimodal_context }
    Side effect: INSERT multimodal_context
 
Compensation (on failure):
  StoreResults.compensate: SET analysis_status='failed'
  DownloadMedia.compensate: DELETE S3 object
 
Deterministic Jitter: jitter_ms = fnv1a_hash(workflow_run_id + attempt) % 2000

4.3 FederationNegotiationWorkflow (Phase 2)
YAML
ID: federation-neg-{negotiation_id}
Task Queue: federation
Execution Timeout: 86400s (24h)
Idempotency: federation_id + negotiation_type + idempotency_key
 
State Machine:
  PROPOSED → EVALUATING → COUNTER_PROPOSED (loop max_rounds) → ACCEPTED/REJECTED/EXPIRED
  ACCEPTED → EXECUTING → EXECUTED/FAILED
  FAILED → COMPENSATING → TERMINAL
 
Activities:
  ValidateFederation:
    Timeout: 5s
    Checks: federation.status == 'active', permission_type in permissions_granted, daily negotiation count < max
 
  SendProposal:
    Timeout: 10s
    Side effect: Write to federation_negotiations, enqueue SQS message to partner workspace
    Compensation: Mark negotiation as 'failed'
 
  EvaluateProposal (runs on RESPONDER workspace):
    Timeout: 300s (may involve Brain reasoning)
    Logic:
      Load responder preferences, calendar, constraints
      Brain decides: accept / counter / reject
      IF auto-acceptable (slot free, no conflicts): auto-accept
      IF ambiguous: generate counter-proposal
      IF unacceptable: reject with reason
      Responder autonomy_level governs human-in-loop requirement
 
  ExecuteAgreement:
    Timeout: 60s
    Two-Phase Commit:
      Phase 1: Simulate on both workspaces (e.g., dry-run calendar create)
      Phase 2: Commit on both workspaces
      IF either simulation fails: abort, notify both parties
    Compensation: Reverse committed actions on both sides
 
Privacy Rules (OPA-enforced):
  1. NEVER share raw memory_items across boundaries
  2. NEVER share raw conversation content
  3. ONLY share typed proposal/response payloads matching permission schemas:
     calendar_query → { available_slots: [{ start, end }] }
     calendar_propose → { proposed_time, duration, topic_summary }
     meeting_negotiate → { options: [{ time, location }], constraints }
     task_delegate → { task_description, deadline, acceptance_criteria }
     knowledge_share → { topic, summary, source_type }
     status_query → { query_type, response: boolean|enum }
  4. All payloads schema-validated before transmission
  5. Every cross-workspace access logged in federation_audit_log

4.4 PurchaseWorkflow (Phase 6)
YAML
ID: purchase-{transaction_id}
Task Queue: purchases
Execution Timeout: 3600s
Idempotency: idempotency_key (provided by caller)
 
Activities:
  ValidateMerchant:
    Timeout: 5s
    Logic: Check approved_merchants. IF blocked → DENY. IF pending → create approval request.
    
  CheckBudget:
    Timeout: 5s
    Checks (ALL must pass):
      amount <= wallet.single_tx_limit_cents
      wallet.spent_today_cents + amount <= wallet.daily_limit_cents
      wallet.spent_this_month_cents + amount <= wallet.monthly_limit_cents
      amount < wallet.hard_block_cents
      wallet.balance_cents >= amount
    IF any fails → DENY with specific error code
 
  RequestApproval (conditional: amount > approval_threshold_cents):
    Timeout: 1800s (30 minutes)
    Logic:
      Send approval request via WhatsApp/iMessage
      Include: merchant, amount, description, category
      Generate HMAC-SHA256 approval token (TTL=30min, nonce-unique)
      Wait for user response (YES/NO or approval link click)
      IF timeout → DENY with WALLET_APPROVAL_TIMEOUT
 
  ProcessPayment:
    Timeout: 30s
    Two-Phase Commit:
      Phase 1 (Simulate): Create Stripe PaymentIntent capture_method=manual
      Phase 2 (Commit): Capture PaymentIntent
    Side effects:
      Deduct wallet.balance_cents atomically (SELECT FOR UPDATE)
      Increment spent_today_cents and spent_this_month_cents
      Store payment_provider_ref
    Compensation: Stripe refund + credit wallet.balance_cents
 
  CaptureReceipt:
    Timeout: 15s
    Logic: Fetch receipt URL → store in S3 → create message_attachment → auto-categorize expense
 
Compensation Chain (reverse order):
  CaptureReceipt.compensate: delete S3 receipt (optional, keep for audit)
  ProcessPayment.compensate: Stripe refund, credit wallet
  RequestApproval.compensate: notify user of cancellation
  CheckBudget.compensate: no-op
  ValidateMerchant.compensate: no-op

  FirstValue: Ask "What's one thing I can help you with right now?" Execute task with extra explanation.
  Complete: Mark onboarding done. Begin normal operation.
 
First-value design:
  The goal is user experiences real value within 5 minutes.
  Brevio executes the task with verbose explanation of what it's doing and why.
  After completion: "That's what I can do. Want me to keep going?"

13. RUNBOOKS
13.1 Federation Incident Runbook
1. Alert: federation_privacy_violation > 0
2. IMMEDIATELY: Halt all active federation negotiations (set status='suspended' globally)
3. Identify: Query federation_audit_log for the violating event
4. Assess: Determine what data crossed boundary, which workspaces affected
5. Contain: Revoke specific federation. Notify both workspace owners.
6. Fix: Patch OPA policy or schema validation gap
7. Verify: Run full federation policy test suite
8. Resume: Re-enable federation system after fix verified
9. Post-incident: Update threat model, add regression test

13.2 Wallet Reconciliation Runbook
1. Alert: wallet_balance_desync OR manual report of incorrect balance
2. Pause wallet: SET digital_wallets.is_active = false for affected workspace
3. Audit: Compare wallet.balance_cents with SUM(wallet_transactions) grouped by state
4. Compare: Cross-reference with Stripe dashboard (payments, refunds)
5. Identify: Locate missing or duplicate transactions
6. Fix: INSERT corrective transaction with category='reconciliation'
7. Verify: New balance matches Stripe + transaction log
8. Resume: SET digital_wallets.is_active = true
9. Post-incident: Add monitoring for balance drift > $1

13.3 WhatsApp Window Expiry Runbook
1. Brevio attempts free-form message, gets WHATSAPP_WINDOW_EXPIRED
2. Automatic: Fall back to approved template message
3. IF no appropriate template exists:
   a. Queue message for delivery on next user-initiated conversation
   b. Log as brevio_whatsapp_window_expires_total
4. IF critical (approval request, payment confirmation):
   a. Send using 'utility' template: "Brevio needs your attention: {{1}}"
   b. Include deep link to web dashboard for full context
5. Monitor: If window expiry rate > 20%, consider more proactive templates

13.4 Edge Agent Recovery Runbook
1. Alert: Edge Agent offline for > 30 minutes
2. Check: Is user's device powered on? (no telemetry = likely device off)
3. IF device reachable but agent crashed:
   a. Remote restart via OS-level service manager (launchd/systemd)
   b. Verify WebSocket reconnection within 30s
   c. Sync offline_task_queue
4. IF certificate expired:
   a. Notify user via WhatsApp: "Your Mac assistant needs reconnection"
   b. Provide re-registration deep link
   c. Revoke old certificate
5. IF persistent failure:
   a. Collect agent logs (if available)
   b. Suggest agent reinstallation
   c. Escalate to engineering if pattern affects multiple users

B.3 EDGE AGENT WEBSOCKET PROTOCOL SPECIFICATION
The Control service maintains persistent WebSocket connections to Edge Agents (macOS, Windows, Linux). This section defines the complete binary/text protocol: message types, framing, authentication, heartbeat, reconnect logic, and error handling.
🔴 RULE: All Edge Agent WebSocket connections MUST use TLS 1.3 with mutual authentication (mTLS). Connections without valid client certificates are rejected at the TCP layer.
B.3.1 Connection Establishment
URL: wss://ws.brevio.ai:8443/v1/edge/connect

Connection headers (client → server):
  Authorization: Bearer {edge_agent_jwt}    // short-lived, 15min TTL
  X-Edge-Agent-ID: {uuid}
  X-Edge-Platform: macos|windows|linux
  X-Edge-Version: 2.0.0
  X-Edge-Workspace-ID: {workspace_uuid}

Auth flow:
  1. Edge Agent presents mTLS client cert (issued by CA at brevio/{env}/edge/ca_cert)
  2. Control validates cert against CA. Reject if expired, revoked, or workspace mismatch.
  3. Control issues edge_agent_jwt (HS256, signed with EDGE_CA_KEY, 15min TTL)
  4. Agent uses jwt in Authorization header
  5. On jwt expiry: agent re-requests via POST /v1/edge/token (mTLS only, no jwt needed)
B.3.2 Message Format (JSON over WebSocket text frames)
{
  "type": "<MessageType>",
  "message_id": "<uuid>",           // client-generated for requests, server-generated for events
  "timestamp": "<ISO8601>",
  "workspace_id": "<uuid>",
  "edge_agent_id": "<uuid>",
  "payload": { ... }               // type-specific, see below
}
B.3.3 Message Type Registry
B.3.4 Heartbeat Logic
Interval: every EDGE_HEARTBEAT_INTERVAL_S (default: 30s)
Sequence: monotonically increasing integer per connection

Server sends HEARTBEAT every 30s.
Edge must respond with HEARTBEAT_ACK within 10s.
If no ACK within 10s: server increments missed_heartbeats counter.
If missed_heartbeats >= 3: server marks edge_agent.is_online = false, closes connection.

Edge sends HEARTBEAT if no message received for 45s (keepalive from client side).
Server responds with HEARTBEAT_ACK.
B.3.5 Reconnect Logic (Edge Client)
Reconnect strategy: exponential backoff with deterministic jitter
  attempt 1: wait 1s + fnv1a_hash(edge_agent_id + attempt) % 1000 ms
  attempt 2: wait 2s + jitter
  attempt 3: wait 4s + jitter
  attempt N: min(2^(N-1), 60) seconds + jitter
Max reconnect window: EDGE_RECONNECT_TIMEOUT_S (300s)

On successful reconnect:
  1. Send CAPABILITY_REPORT
  2. Trigger EdgeSyncWorkflow via POST /v1/edge/sync (mTLS)
  3. Resume receiving TASK_DISPATCH messages
B.3.6 TASK_DISPATCH Payload Schema
{
  "task_id": "<uuid>",
  "skill_id": "<string>",
  "input": { ... },              // skill-specific input, schema from connector_tools.input_schema
  "priority": 1-10,             // 1=highest; drives execution order
  "offline_tier": 1-4,          // see B.4 for tier definitions
  "expires_at": "<ISO8601>",   // task invalid after this time; discard without executing
  "idempotency_key": "<string>" // for deduplication on reconnect
}

B.6 FAST-PATH PATTERN MATCHING ALGORITHM
The V10 Base document specifies fast_path_routes with query_pattern and confidence_threshold but does not define how user messages are matched against patterns. This section provides the authoritative algorithm.
B.6.1 Matching Pipeline
func MatchFastPath(userMsg string, routes []FastPathRoute) *FastPathMatch {
    // Step 1: Normalize user message
    normalized := strings.ToLower(strings.TrimSpace(userMsg))
    normalized = removeFillerWords(normalized)  // strip 'please', 'can you', 'hey brevio'

    // Step 2: Check exact hash match (O(1))
    hash := fnv1a64(normalized)
    if route := routesByHash[hash]; route != nil && route.IsActive {
        return &FastPathMatch{Route: route, Confidence: 1.0, MatchType: 'exact'}
    }

    // Step 3: Template variable substitution match
    for _, route := range routes {
        if ok, vars := matchTemplate(normalized, route.QueryPattern); ok {
            return &FastPathMatch{Route: route, Confidence: 0.95, Vars: vars, MatchType: 'template'}
        }
    }

    // Step 4: Embedding similarity (only if Redis cache has embeddings)
    if embedding := getQueryEmbedding(normalized); embedding != nil {
        best := findMostSimilar(embedding, routeEmbeddings)
        if best.CosineSimilarity >= route.ConfidenceThreshold {
            return &FastPathMatch{Route: best.Route, Confidence: best.CosineSimilarity, MatchType: 'semantic'}
        }
    }

    return nil  // no fast-path match, fall through to classifier
}
B.6.2 Template Pattern Syntax
// Pattern: 'what is my next {event_type}'
// Matches: 'what is my next meeting', 'what is my next standup'
// {variable} tokens capture any single word or short phrase (max 32 chars)

func matchTemplate(msg string, pattern string) (bool, map[string]string) {
    // Convert pattern to regex: {var} => (?P<var>[\w\s]{1,32})
    regex := templateToRegex(pattern)
    m := regex.FindStringSubmatch(msg)
    if m == nil { return false, nil }
    vars := extractNamedGroups(regex, m)
    return true, vars
}
B.6.3 Filler Words to Strip

    "brevio", "quick question", "just wondering", "i was wondering",

B.9 RATE LIMITING ENFORCEMENT IMPLEMENTATION
The unified_rate_limits table (defined in Migration 004) stores limits but does not specify the enforcement algorithm. This section defines the sliding window algorithm, middleware integration, and Redis key structure.
B.9.1 Algorithm: Sliding Window Counter
// Implementation: Redis sorted set sliding window
// Per request, the middleware executes this Lua script atomically:

local key = KEYS[1]              -- rate:ws:{workspace_id}:svc:{service}:min
local window = tonumber(ARGV[1]) -- 60 (seconds)
local limit = tonumber(ARGV[2])  -- requests_per_minute
local now = tonumber(ARGV[3])    -- unix timestamp in ms
local request_id = ARGV[4]       -- unique per request (uuid)

-- Remove expired entries
redis.call('ZREMRANGEBYSCORE', key, 0, now - (window * 1000))

-- Count current window
local count = redis.call('ZCARD', key)

if count >= limit then
    return {0, count, limit}  -- denied
end

-- Add current request
redis.call('ZADD', key, now, request_id)
redis.call('EXPIRE', key, window + 1)
return {1, count + 1, limit}  -- allowed
B.9.2 Middleware Integration (Gateway)
func RateLimitMiddleware(service string) gin.HandlerFunc {
    return func(c *gin.Context) {
        workspaceID := c.GetString('workspace_id')
        tier := c.GetString('tier')  // from JWT claim

        limits, _ := getLimits(workspaceID, service, tier)

        // Check per-minute
        allowed, current, max := slidingWindow(
            fmt.Sprintf('rate:%s:%s:min', workspaceID, service),
            60, limits.RequestsPerMinute)
        if !allowed {
            c.Header('X-RateLimit-Limit', strconv.Itoa(max))
            c.Header('X-RateLimit-Remaining', '0')
            c.Header('Retry-After', '60')
            c.AbortWithStatusJSON(429, gin.H{'error': 'PLAN_LIMIT_REACHED'})
            return
        }

        // Burst check: if current > burst_limit, apply additional delay
        c.Header('X-RateLimit-Limit', strconv.Itoa(max))
        c.Header('X-RateLimit-Remaining', strconv.Itoa(max - current))
        c.Next()
    }
}

B.13.1 Assignment Algorithm
// Deterministic bucketing — same user always gets same variant for same experiment
func AssignVariant(userID uuid.UUID, experimentID uuid.UUID, trafficSplitPct int) string {
    // 1. Create stable hash from user + experiment
    hash := fnv1a32(userID.String() + ':' + experimentID.String())

    // 2. Map to 0-99 bucket
    bucket := hash % 100

    // 3. traffic_split_pct = % assigned to variant
    //    e.g., 50% split: bucket 0-49 = variant, 50-99 = control
    if bucket < trafficSplitPct {
        return 'variant'
    }
    return 'control'
}

// On first request for user in an experiment:
// 1. Check experiment_assignments for existing assignment (return if found)
// 2. Check experiment.status == 'running' (skip if not)
// 3. Call AssignVariant() to determine assignment
// 4. INSERT INTO experiment_assignments (idempotent via UNIQUE(experiment_id, user_id))
// 5. Emit: BREVIO.experiments.user_assigned.v1

B.19 GRACEFUL SHUTDOWN PROCEDURE

// All services implement this shutdown sequence on SIGTERM:

func GracefulShutdown(server *http.Server, db *pgxpool.Pool, redis *redis.Client) {
    // 1. Stop accepting new requests (close HTTP listener)
    ctx, cancel := context.WithTimeout(context.Background(), 25*time.Second)
    defer cancel()
    server.Shutdown(ctx)  // drains in-flight requests, max 25s

    // 2. Stop Temporal workflow workers (drain executing activities)
    temporalWorker.Stop()  // blocks until current activities complete, max 20s

    // 3. Close database connection pool (after in-flight queries complete)
    db.Close()

    // 4. Close Redis connections
    redis.Close()

    // 5. Flush telemetry
    tracerProvider.Shutdown(ctx)

    logger.Info('graceful shutdown complete')
}
